import {
  Controller,
  Get,
  Post,
  Body,
  Patch,
  Param,
  Delete,
  Query,
  UseGuards,
  Request,
  ParseUUIDPipe,
} from '@nestjs/common';
import {
  ApiTags,
  ApiOperation,
  ApiResponse,
  ApiBearerAuth,
  ApiQuery,
  ApiParam,
  ApiBody,
} from '@nestjs/swagger';
import { ConfigService } from '@nestjs/config';
import { User } from './entities/user.entity';
import { EmailService } from '../email/email.service';
import { UsersService } from './users.service';
import { CreateUserDto } from './dto/create-user.dto';
import { UpdateUserDto } from './dto/update-user.dto';

    console.log(`   üìù Dados: { email: ${createUserDto.email}, firstName: ${createUserDto.firstName}, role: ${createUserDto.role} }`);
    
    try {
      const newUser = await this.usersService.create(createUserDto);
      
      console.log(`‚úÖ [CREATE_USER] Usu√°rio criado com sucesso - ID: ${newUser.id}, Email: ${newUser.email}`);
      
      return newUser;
    } catch (error) {
      console.log(`‚ùå [CREATE_USER] Erro ao criar usu√°rio: ${error.message}`);
      throw error;

  }

  @Post('diretor-autologin')
  @Roles(UserRole.ADMINISTRADOR)
  @ApiOperation({ summary: 'Criar Diretor com login autom√°tico (Apenas Administradores)' })
  @ApiBody({
    schema: {
      type: 'object',
      required: ['email', 'firstName', 'lastName'],
      properties: {
        email: { type: 'string', format: 'email' },
        firstName: { type: 'string' },
        lastName: { type: 'string' },
      },
    },
  })
  @ApiResponse({ status: 201, description: 'Diretor criado com login autom√°tico' })
  async createDirectorAutologin(@Body() body: { email: string; firstName: string; lastName: string }, @Request() req) {
    const currentUser = req.user;
    const { email, firstName, lastName } = body;

    console.log(`üõ†Ô∏è [CREATE_DIRETOR_AUTOLOGIN] Solicita√ß√£o por: ${currentUser.email} (${currentUser.role})`);
    console.log(`   üìß Dados: { email: ${email}, firstName: ${firstName}, lastName: ${lastName} }`);

    const newUser = await this.usersService.create({
      email,
      firstName,
      lastName,
      role: UserRole.DIRETOR as any,
    } as CreateUserDto);

    const activated = await this.usersService.activateUser(newUser.id);
    const { token } = await this.usersService.enableAutoLogin(activated.id);

    const frontendBase = this.configService.get<string>('FRONTEND_URL', 'http://localhost:3005');
    const autologinUrl = `${frontendBase.replace(/\/$/, '')}/autologin/${token}`;

    // Enviar e-mail com link de autologin
    try {
      const subject = 'Acesso ao Sistema - Login Autom√°tico';
      const html = `
        <div style="font-family: Arial, sans-serif; color: #1f2937;">
          <h2>Ol√°, ${firstName}!</h2>
          <p>Foi criada uma conta de Diretor para voc√™ no sistema.</p>
          <p>Para acessar imediatamente, clique no bot√£o abaixo. Voc√™ entrar√° automaticamente no sistema, sem precisar de senha:</p>
          <p style=\"margin:24px 0;\">
            <a href=\"${autologinUrl}\" style=\"display:inline-block;padding:12px 18px;background:#3b82f6;color:#fff;text-decoration:none;border-radius:8px;\">
              Entrar automaticamente
            </a>
          </p>
          <p>Se o bot√£o n√£o funcionar, copie e cole este link no seu navegador:</p>
          <p style=\"word-break:break-all;color:#2563eb;\">${autologinUrl}</p>
        </div>
      `;
      await this.emailService.sendEmail({ to: email, subject, html, text: autologinUrl });
    } catch (e: any) {
      console.log(`Falha ao enviar e-mail de autologin: ${e?.message}`);
    }

    const userSafe: Partial<User> = {
      id: activated.id,
      email: activated.email,
      firstName: activated.firstName,
      lastName: activated.lastName,
      role: activated.role,
      status: activated.status,
      emailVerified: activated.emailVerified,
      lastLogin: activated.lastLogin,
      createdAt: activated.createdAt,
      updatedAt: activated.updatedAt,
      autoLoginEnabled: activated.autoLoginEnabled,
    } as any;

    console.log(`‚úÖ [CREATE_DIRETOR_AUTOLOGIN] Diretor criado e autologin habilitado: ${activated.email}`);

    return {
      user: userSafe,
      autologinToken: token,
      autologinUrl,
    };
  }

  @Get(':id')
  @Roles(UserRole.ADMINISTRADOR) // ‚úÖ ADICIONAR ROLE
  @ApiOperation({ summary: 'Buscar usu√°rio por ID' })

      return user;
    } catch (error) {

      throw error;
    }
  }

  @Patch(':id')
  @Roles(UserRole.ADMINISTRADOR) // ‚úÖ CORRIGIDO

  @ApiParam({ name: 'id', description: 'ID do usu√°rio' })

    } catch (error) {

      throw error;

  }
